class HatredVision : StaticEventHandler
{
	override void WorldLoaded(WorldEvent e)
	{
		PlayerInfo p = players[consoleplayer];
		if (p.mo is "NotimPortant")
		{
			Shader.SetEnabled( p, "SinCity", true );
		}
		else
		{
			Shader.SetEnabled( p, "SinCity", False );
		}
	}
}
