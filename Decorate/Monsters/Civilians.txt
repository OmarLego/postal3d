ACTOR Civilian1 : Postal3dMonsterBase
{
	Health 40
	+SHOOTABLE
	+SOLID
	+CANPASS
	+CANPUSHWALLS
	+CANUSEWALLS
	+ACTIVATEMCROSS
	  radius 14
  height 56
  PainChance 160
Speed 3
  obituary "%o was telefucked by an innocent"
  seesound "act1_civspot"
  attacksound ""
  deathsound "act1_death"
  activesound "act1_civspot"
  painSound "act1_hit"
    +FLOORCLIP
  +DONTOVERLAP
  +RANDOMIZE
  +NOSPLASHALERT
  Mass 60
	Scale 0.65
	Species "Civilians"
	States
	{
  Spawn:
  CIV1 F 4 A_Look
  Loop
  See:
  CIV1 F 11
  Ruuuun:
  CIV1 AAAABBBBCCCCDDDDAAAABBBBCCCCDDDDAAAABBBBCCCCDDDDAAAABBBBCCCCDDDDAAAABBBBCCCCDDDDAAAABBBBCCCCDDDD 1 A_Wander
  CIV1 A 0 A_Jump(88,"Stand","NOPLACELIKEHOME")
  Loop
  Stand:
  CIV1 FF 40
  Goto See
  NOPLACELIKEHOME:
  CIV1 A 0 A_PlaySound("act1_civspot")
	Goto Ruuuun
  Pain:
  CIV1 E 4 A_Pain
  CIV1 E 4 A_Jump(60,"NOPLACELIKEHOME")
  Goto Ruuuun
    Death:
    Death.Bullet:
        TNT1 A 0 A_ChangeFLag("NODROPOFF", 0)
        TNT1 A 0 A_Scream
        TNT1 A 0 A_NoBlocking
        TNT1 A 0 A_Jump(256, "DieNormal", "Crawl")
    Death.Kick:
        TNT1 A 0 A_ChangeFLag("NODROPOFF", 0)
        TNT1 A 0 A_Scream
        TNT1 A 0 A_NoBlocking
        TNT1 A 0 A_Jump(256, "DieKick")
    Death.Explosive:
        TNT1 A 0 A_ChangeFLag("NODROPOFF", 0)
        TNT1 A 0 A_Scream
        TNT1 A 0 A_NoBlocking
        TNT1 A 0 A_JumpIf(health > -10, "DieNormal")
        TNT1 A 0 A_Jump(256, "DieX")
    XDeath:
        TNT1 A 0 A_XScream
        TNT1 A 0 A_NoBlocking
        TNT1 A 0 A_FaceTarget
        TNT1 A 0 A_Jump(256, "DieX")
    DieNormal:
        CIV1 GHIJ 6
		TNT1 A 0 A_SpawnItemEx("GrowingBloodPool")
        CIV1 J -1
    DieKick:
        TNT1 A 0 ThrustThingZ(0,10,0,1)
        TNT1 A 0 A_FaceTarget
        TNT1 A 0 A_Recoil(random(5, 7))
        CIV1 GHIJ 6
		TNT1 A 0 A_SpawnItemEx("GrowingBloodPool")
        CIV1 L -1
    DieX:
        TNT1 A 0 A_SpawnItemEx("MeatDeath")
		TNT1 A 0 A_SpawnItemEx("GrowingBloodPool")
		Stop
    Raise:
        CIV1 JIHG 6
        Goto See
	Crawl:
        TNT1 A 0 A_SpawnItem("DyingZombieman")
		Stop
    Death.fire:
	Death.Flames:
	Death.burn:
	Death.flame:
		TNT1 A 0 A_NoBlocking
		TNT1 A 0 A_SpawnItem("BurningActor1")
		Stop
	}
}
