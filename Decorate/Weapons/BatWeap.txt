ACTOR BatWeap : Postal3dWeaponBase
{
    +WEAPON.MELEEWEAPON
    +WEAPON.NOALERT
    +WEAPON.NOAUTOAIM
    +WEAPON.NOAUTOFIRE
	Tag "Bat"
	Inventory.PickupSound "weapons/bat/pickup"
	Inventory.PickupMessage "You got the Bat!"
  States
  { 
	Ready:
	TNT1 A 0 A_JumpIfInventory("Kicking",1,"Kick")
	BBAT A 1 A_Weaponready (WRF_ALLOWRELOAD)
	loop
	
	Select:
	TNT1 A 0 A_PlaySound ("WEPSEL",CHAN_AUTO)
	BBAT A 1 A_Raise
	Goto Ready
	
	Deselect:
	BBAT A 1 A_Lower
	loop
	
	Spawn:
	BATP A 1
	loop
	
	Fire:
		BBAT BC 2
		TNT1 A 5
		TNT1 A 0 A_PlaySound("weapons/bat/swing",CHAN_AUTO)
		BBAT JHF 1
		TNT1 A 0 A_FireCustomMissile("BatStrike", 0, 0, 0, -3)
		BBAT D 1
		TNT1 A 5
		BBAT CB 2
		Goto Ready
	AltFire:
		BBAT BC 2
		TNT1 A 0 A_PlaySound("weapons/bat/swing",CHAN_AUTO)
		BBAT DF 1
		TNT1 A 0 A_FireCustomMissile("BatStrike", 0, 0, 0, -3)
		BBAT GFDCB 5
		Goto Ready
  }
}

actor BatStrike: KickAttack
{
    DamageType Kick
    Damage (random(64, 64))
    Speed 30

    States
    {
    Spawn:
        TNT1 A 1
        stop

    Death:
        TNT1 A 0 A_PlaySound("weapons/bat/hitwall",CHAN_AUTO)
        TNT1 A 0 A_GiveToTarget("KickHasHit", 1)
        TNT1 A 0 A_Quake(2, 6, 0, 320)
		MPUF A 3
		MPUF BCD 3
        stop
    XDeath:
        TNT1 A 0 A_PlaySound("weapons/bat/hit",CHAN_AUTO)
        TNT1 A 0 A_GiveToTarget("KickHasHit", 1)
        TNT1 A 0 A_Quake(2, 6, 0, 320)
		MPUF A 3
		MPUF BCD 3
        stop
    }
}
