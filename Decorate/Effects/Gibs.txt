Actor PrldGib
{
	
    var int user_rotationspeed;

    +NOBLOCKMAP
    +NOTELEPORT
    +FORCEXYBILLBOARD
    +CLIENTSIDEONLY
    +MISSILE
    -BLOODSPLATTER
    +THRUACTORS
	
    Radius 8
    Height 8
    Speed 10
    Gravity 0.6
    Mass 4
    BounceFactor 0.01
	
    +DOOMBOUNCE
    +MOVEWITHSECTOR
    +ROLLSPRITE
    +ROLLCENTER
    -RANDOMIZE
	
	XScale 0.5
	YScale 0.58
    States
    {
    Spawn:
		TNT1 A 0
        TNT1 A 0 A_SpawnItem("BloodSpot", 0, 0, 0, 1)
        TNT1 A 0 A_SetUserVar("user_rotationspeed", random(30, 50))
        TNT1 A 0 A_SetUserVar("user_rotationspeed", user_rotationspeed * randompick(-1, 1))
        TNT1 A 0 A_JumpIf(waterlevel >= 1, "Underwater")
        TNT1 A 0 A_Jump(256, "Fly")
    Underwater:
        TNT1 A 0
        Stop
    Lying:
        "----" A 1
		TNT1 A 0 A_QueueCorpse
		"----" A -1
		Stop
    }
}
