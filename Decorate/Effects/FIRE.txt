  //Scale 1.2

  +NOBLOCKMAP
  +NOCLIP
  States
  {
	Spawn:
		TNT1 A 0 A_SetScale(0.4)
		FLME A 3 BRIGHT
		TNT1 A 0 A_SetScale(0.6)

		TNT1 A 0 A_SetScale(0.8)

		TNT1 A 0 A_SetScale(1.0)

		FLME ABCDABCDABCDABCDABCDABCDABCD 2 BRIGHT A_Explode(5,16,0, 0, 0, 0, 0, 0, "Fire")
		TNT1 A 0 A_Jump(168, "Death")
		Loop
	Death:
		FLME ABCDAB 3 bright 
		FLME C 1 bright A_FadeOut(0.2)
		FLME C 1 bright A_SetScale(1.0)
		FLME C 1 bright A_FadeOut(0.2)
		FLME D 1 bright A_SetScale(0.8)
		FLME D 1 A_FadeOut(0.2)

		FLME A 1 A_FadeOut(0.2)
		FLME A 1 A_SetScale(0.4)
		FLME A 1 A_FadeOut(0.2)

  }
}
